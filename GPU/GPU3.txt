D = Depend
A = ADR or ID

0 = Fbuffer_select
1 = Zbuffer_select
2 = execute
3 = end
4 = wait
5 = flush cache buffer or texture
6 = buffer size init

13 = RAM <- L1_SPM-I

15 = L2_SPM <- L1_SPM-I
