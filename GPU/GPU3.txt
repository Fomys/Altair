0xCCDD DDDD AAAA AAAA
C = CMD
D = Depend
A = ADR or ID

0 = Fbuffer_select
1 = Zbuffer_select
2 = Load_DATA (L2_SPM/RAM -> L1_SPM)
3 = Store_DATA (L1_SPM -> L2_SPM/RAM )
4 = flush cache buffer
5 = flush cache texture
6 = execute
7 = end

0xCCDD Daaa AAAA AAAA (2kib)
8 = RAM -> L2_SPM
9 = L2_SPM -> RAM

0xCCDD Daaa AAAA AAAA (16kib)
10 = RAM -> 1T-SRAM
11 = 1T-SRAM -> RAM

------------------------------------

CMD 0: select_Fbuffer
CMD 1: select_Zbuffer
CMD 8: Load_DATA vertexshader SPM2
CMD 8: Load_DATA pixelshader SPM2
CMD 8: Load_DATA gen SPM2 (texture select /matrix/other)
CMD 8: Load_DATA Model
CMD 10: Load_DATA texture
